### Задание 2

```
pipeline {
    agent any
    stages {
        stage('Git') {
            steps {
                git branch: 'main', url: 'https://github.com/Night-N/ci-cd-materials/'
            }
            }
        stage('Go test') {
            steps {
                sh "go test ."
            }
            }
        stage('Docker test') {
            steps {
                sh "docker build ."
            }
            }
        }
    }
```
